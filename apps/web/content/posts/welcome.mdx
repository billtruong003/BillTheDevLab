  language="glsl"
  fileName="volumetric-fog.frag"
  code={`uniform float u_time;
uniform vec3 u_fogColor;
uniform float u_density;

float fogFactor(float dist) {
  return 1.0 - exp(-u_density * dist * dist);
}

void main() {
  float dist = length(v_worldPos - u_cameraPos);
  float fog = fogFactor(dist);
  vec3 color = mix(v_color, u_fogColor, fog);
  gl_FragColor = vec4(color, 1.0);
}`}
  highlightLines={[6, 7]}
